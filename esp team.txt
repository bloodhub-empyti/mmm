getgenv().ESPEnabled = true

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Camera = workspace.CurrentCamera
local LocalPlayer = Players.LocalPlayer

local ESP = {}
ESP.__index = ESP

function ESP.new()
    local self = setmetatable({}, ESP)
    self.espCache = {}
    return self
end

function ESP:createDrawing(type, props)
    local obj = Drawing.new(type)
    for i,v in pairs(props or {}) do
        obj[i] = v
    end
    return obj
end

function ESP:createComponents()
    return {
        Box = self:createDrawing("Square", {
            Thickness = 1,
            Color = Color3.fromRGB(255,255,255),
            Filled = false,
            Visible = false
        }),

        NameLabel = self:createDrawing("Text", {
            Size = 11,
            Center = true,
            Outline = true,
            Color = Color3.fromRGB(255,255,255),
            Visible = false
        }),

        DistanceLabel = self:createDrawing("Text", {
            Size = 11,
            Center = true,
            Outline = true,
            Color = Color3.fromRGB(255,255,255),
            Visible = false
        }),

        HealthText = self:createDrawing("Text", {
            Size = 11,
            Center = true,
            Outline = true,
            Color = Color3.fromRGB(0,255,0),
            Visible = false
        }),

        ItemLabel = self:createDrawing("Text", {
            Size = 11,
            Center = true,
            Outline = true,
            Color = Color3.fromRGB(255,255,0),
            Visible = false
        }),

        SkeletonLines = {}
    }
end

local bodyConnections = {
    R15 = {
        {"Head","UpperTorso"},
        {"UpperTorso","LowerTorso"},
        {"UpperTorso","LeftUpperArm"},
        {"LeftUpperArm","LeftLowerArm"},
        {"LeftLowerArm","LeftHand"},
        {"UpperTorso","RightUpperArm"},
        {"RightUpperArm","RightLowerArm"},
        {"RightLowerArm","RightHand"},
        {"LowerTorso","LeftUpperLeg"},
        {"LeftUpperLeg","LeftLowerLeg"},
        {"LeftLowerLeg","LeftFoot"},
        {"LowerTorso","RightUpperLeg"},
        {"RightUpperLeg","RightLowerLeg"},
        {"RightLowerLeg","RightFoot"}
    },

    R6 = {
        {"Head","Torso"},
        {"Torso","Left Arm"},
        {"Torso","Right Arm"},
        {"Torso","Left Leg"},
        {"Torso","Right Leg"}
    }
}

function ESP:updateComponents(c, character, player)
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")

    if not humanoid or not hrp or humanoid.Health <= 0 then
        self:hideComponents(c)
        return
    end

    local rootPos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
    if not onScreen then
        self:hideComponents(c)
        return
    end

    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end

    local distance = math.floor(
        (LocalPlayer.Character.HumanoidRootPart.Position - hrp.Position).Magnitude
    )

    local scale = 1 / rootPos.Z * 100
    local width = 25 * scale
    local height = 30 * scale

    c.Box.Size = Vector2.new(width, height)
    c.Box.Position = Vector2.new(rootPos.X - width/2, rootPos.Y - height/2)
    c.Box.Visible = true

    c.NameLabel.Text = "["..player.Name.."]"
    c.NameLabel.Position = Vector2.new(rootPos.X, rootPos.Y - height/2 - 15)
    c.NameLabel.Visible = true

    c.DistanceLabel.Text = "["..distance.."M]"
    c.DistanceLabel.Position = Vector2.new(rootPos.X, rootPos.Y + height/2 + 15)
    c.DistanceLabel.Visible = true

    local hpPercent = math.floor((humanoid.Health / humanoid.MaxHealth) * 100)
    c.HealthText.Text = "[HP: "..hpPercent.."%]"
    c.HealthText.Position = Vector2.new(rootPos.X, rootPos.Y + height/2 + 30)
    c.HealthText.Color = Color3.fromRGB(255 - hpPercent*2.55, hpPercent*2.55, 0)
    c.HealthText.Visible = true

    local tool = player.Backpack:FindFirstChildOfClass("Tool")
        or character:FindFirstChildOfClass("Tool")

    if tool then
        c.ItemLabel.Text = "[Holding: "..tool.Name.."]"
    else
        c.ItemLabel.Text = "[Holding: None]"
    end

    c.ItemLabel.Position = Vector2.new(rootPos.X, rootPos.Y + height/2 + 45)
    c.ItemLabel.Visible = true

    local connections = bodyConnections[humanoid.RigType.Name] or {}

    for _, connection in ipairs(connections) do
        local partA = character:FindFirstChild(connection[1])
        local partB = character:FindFirstChild(connection[2])

        if partA and partB then
            local key = connection[1].."-"..connection[2]

            local line = c.SkeletonLines[key]
                or self:createDrawing("Line", {Thickness = 1, Color = Color3.fromRGB(255,255,255)})

            local posA, visA = Camera:WorldToViewportPoint(partA.Position)
            local posB, visB = Camera:WorldToViewportPoint(partB.Position)

            if visA and visB then
                line.From = Vector2.new(posA.X, posA.Y)
                line.To = Vector2.new(posB.X, posB.Y)
                line.Visible = true
                c.SkeletonLines[key] = line
            else
                line.Visible = false
            end
        end
    end
end

function ESP:hideComponents(c)
    c.Box.Visible = false
    c.NameLabel.Visible = false
    c.DistanceLabel.Visible = false
    c.HealthText.Visible = false
    c.ItemLabel.Visible = false

    for _, line in pairs(c.SkeletonLines) do
        line.Visible = false
    end
end

function ESP:removeEsp(player)
    local c = self.espCache[player]
    if not c then return end

    c.Box:Remove()
    c.NameLabel:Remove()
    c.DistanceLabel:Remove()
    c.HealthText:Remove()
    c.ItemLabel:Remove()

    for _, line in pairs(c.SkeletonLines) do
        line:Remove()
    end

    self.espCache[player] = nil
end

local esp = ESP.new()

RunService.RenderStepped:Connect(function()

    if not getgenv().ESPEnabled then
        for _, comp in pairs(esp.espCache) do
            esp:hideComponents(comp)
        end
        return
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            if player.Character then
                if not esp.espCache[player] then
                    esp.espCache[player] = esp:createComponents()
                end
                esp:updateComponents(esp.espCache[player], player.Character, player)
            else
                if esp.espCache[player] then
                    esp:hideComponents(esp.espCache[player])
                end
            end
        end
    end
end)

Players.PlayerRemoving:Connect(function(player)
    esp:removeEsp(player)
end)
